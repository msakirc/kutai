```python
    async def handle_get_logs(request: web.Request) -> web.Response:
        fn = request.query.get("file", "")
        path = _safe_filename(log_dir, fn)
        if path is None:
            return web.json_response({"error": "bad file"}, status=400)
        n = int(request.query.get("lines", "1000"))

        # Streaming: count total lines and collect last N
        # Uses a ring buffer to avoid loading all lines into memory
        total = 0
        ring: list[str] = []
        ring_start = 0
        with open(path, "r", encoding="utf-8") as fh:
            for raw in fh:
                raw = raw.rstrip("\n\r")
                if not raw:
                    continue
                total += 1
                if len(ring) < n:
                    ring.append(raw)
                else:
                    ring[ring_start] = raw
                    ring_start = (ring_start + 1) % n
        # Unwind ring buffer into order
        if len(ring) == n:
            lines = ring[ring_start:] + ring[:ring_start]
        else:
            lines = ring
        return web.json_response({"lines": lines, "total": total})

    async def handle_tail(request: web.Request) -> web.Response:
        fn = request.query.get("file", "")
        path = _safe_filename(log_dir, fn)
        if path is None:
            return web.json_response({"error": "bad file"}, status=400)
        after = request.query.get("after", "")
        out: list[str] = []
        with open(path, "r", encoding="utf-8") as fh:
            for raw in fh:
                raw = raw.rstrip("\n\r")
                if not raw:
                    continue
                # Fast substring check for timestamp comparison
                # JSON format guarantees "ts":"..." is near the start
                if after:
                    ts_start = raw.find('"ts"')
                    if ts_start == -1:
                        continue
                    # Extract ts value without full JSON parse
                    val_start = raw.index('"', ts_start + 5) + 1
                    val_end = raw.index('"', val_start)
                    ts_val = raw[val_start:val_end]
                    if ts_val <= after:
                        continue
                out.append(raw)
        return web.json_response({"lines": out})
```

```python
"""Token-based authentication for yazbunu server."""

import json
import os
import secrets
from pathlib import Path

_DEFAULT_DIR = Path.home() / ".yazbunu"


def get_or_create_token(auth_dir: str | None = None) -> str:
    """Return existing token or generate a new one."""
    d = Path(auth_dir) if auth_dir else _DEFAULT_DIR
    d.mkdir(parents=True, exist_ok=True)
    auth_file = d / "auth.json"
    if auth_file.exists():
        try:
            data = json.loads(auth_file.read_text())
            if data.get("token"):
                return data["token"]
        except (json.JSONDecodeError, KeyError):
            pass
    token = secrets.token_urlsafe(32)
    auth_file.write_text(json.dumps({"token": token}, indent=2))
    return token


def validate_token(request_token: str, expected: str) -> bool:
    """Constant-time token comparison."""
    return secrets.compare_digest(request_token, expected)


def build_url(host: str, port: int, token: str, tls: bool = False) -> str:
    """Build the full auth URL."""
    scheme = "https" if tls else "http"
    return f"{scheme}://{host}:{port}?token={token}"
```

```python
import time
from yazbunu.auth import get_or_create_token, validate_token, build_url

_start_time = time.monotonic()
__version__ = "0.2.0"


def create_app(log_dir: str, *, auth_dir: str | None = None,
               require_auth: bool = False, host: str = "0.0.0.0",
               port: int = 9880) -> web.Application:

    token = get_or_create_token(auth_dir)
    url = build_url(host, port, token)

    # --- Auth middleware ---
    @web.middleware
    async def auth_middleware(request: web.Request, handler):
        # /health and /static/ are always open
        if request.path in ("/health", "/auth/qr") or request.path.startswith("/static/"):
            return await handler(request)
        # Root page: allow if token in query (bookmarked URL) — set cookie
        if request.path == "/":
            q_token = request.query.get("token", "")
            if q_token and validate_token(q_token, token):
                resp = await handler(request)
                resp.set_cookie("_yz_token", q_token, httponly=True, max_age=86400 * 30)
                return resp
        if not require_auth:
            return await handler(request)
        # Check query param
        q_token = request.query.get("token", "")
        if q_token and validate_token(q_token, token):
            return await handler(request)
        # Check Authorization header
        auth_header = request.headers.get("Authorization", "")
        if auth_header.startswith("Bearer "):
            h_token = auth_header[7:]
            if validate_token(h_token, token):
                return await handler(request)
        # Check cookie
        c_token = request.cookies.get("_yz_token", "")
        if c_token and validate_token(c_token, token):
            return await handler(request)
        return web.json_response({"error": "unauthorized"}, status=401)

    # --- Health endpoint ---
    async def handle_health(request: web.Request) -> web.Response:
        return web.json_response({
            "status": "ok",
            "version": __version__,
            "uptime_sec": round(time.monotonic() - _start_time, 1),
            "token": token,
            "url": url,
        })

    # ... existing handlers ...

    app = web.Application(middlewares=[auth_middleware])
    app.router.add_get("/health", handle_health)
    # ... existing routes ...
```

```python
    async def handle_stats(request: web.Request) -> web.Response:
        fn = request.query.get("file", "")
        path = _safe_filename(log_dir, fn)
        if path is None:
            return web.json_response({"error": "bad file"}, status=400)

        levels: dict[str, int] = {}
        sources: dict[str, int] = {}
        total = 0
        first_ts: str | None = None
        last_ts: str | None = None

        with open(path, "r", encoding="utf-8") as fh:
            for raw in fh:
                raw = raw.rstrip("\n\r")
                if not raw:
                    continue
                total += 1

                # Extract level via substring (no JSON parse)
                for lv in ("DEBUG", "INFO", "WARNING", "ERROR", "CRITICAL"):
                    if f'"level":"{lv}"' in raw or f'"level": "{lv}"' in raw:
                        levels[lv] = levels.get(lv, 0) + 1
                        break

                # Extract source via substring
                src_start = raw.find('"src"')
                if src_start != -1:
                    try:
                        v_start = raw.index('"', src_start + 6) + 1
                        v_end = raw.index('"', v_start)
                        src_val = raw[v_start:v_end]
                        sources[src_val] = sources.get(src_val, 0) + 1
                    except ValueError:
                        pass

                # Extract timestamp
                ts_start = raw.find('"ts"')
                if ts_start != -1:
                    try:
                        v_start = raw.index('"', ts_start + 5) + 1
                        v_end = raw.index('"', v_start)
                        ts_val = raw[v_start:v_end]
                        if first_ts is None:
                            first_ts = ts_val
                        last_ts = ts_val
                    except ValueError:
                        pass

        # Top 10 sources by count
        top = sorted(sources.items(), key=lambda x: -x[1])[:10]

        return web.json_response({
            "total": total,
            "levels": levels,
            "time_range": {"first": first_ts, "last": last_ts},
            "top_sources": top,
            "indexed": False,
        })
```

```python
    async def handle_context(request: web.Request) -> web.Response:
        fn = request.query.get("file", "")
        path = _safe_filename(log_dir, fn)
        if path is None:
            return web.json_response({"error": "bad file"}, status=400)
        field = request.query.get("field", "")
        value = request.query.get("value", "")
        if not field or not value:
            return web.json_response({"error": "field and value required"}, status=400)

        # Build substring patterns to match without JSON parse
        # Handles both compact and spaced JSON: "field":"value" or "field": "value"
        patterns = [
            f'"{field}":"{value}"',
            f'"{field}": "{value}"',
            f'"{field}":{value}',   # numeric values
            f'"{field}": {value}',
        ]

        out: list[str] = []
        with open(path, "r", encoding="utf-8") as fh:
            for raw in fh:
                raw = raw.rstrip("\n\r")
                if not raw:
                    continue
                if any(p in raw for p in patterns):
                    out.append(raw)
        return web.json_response({"lines": out})
```

```python
                    fh.seek(file_size)

                        raw = raw.rstrip("\n\r")
                        if raw:
                            new_lines.append(raw)
                file_size = new_size
                if new_lines:
                    await ws.send_json({"type": "lines", "lines": new_lines})
        except asyncio.CancelledError:
            pass

        return ws
```

- [ ] **Step 1: Create the Web Worker**

Create `packages/yazbunu/src/yazbunu/static/worker.js`:

```javascript
/**
 * Yazbunu Web Worker — log parsing, indexing, querying, and aggregation.
 *
 * All CPU-heavy work runs here. Main thread only renders.
 *
 * Message protocol:
 *   Main → Worker:
 *     {type:"load", lines:string[], chunk:number}
 *     {type:"clear"}
 *     {type:"filter", query:string}
 *     {type:"stats"}
 *     {type:"aggregate", field:string, bucket:"minute"|"hour"}
 *     {type:"patterns"}
 *     {type:"related", lineIndex:number, windowSec:number}
 *     {type:"fields"}
 *
 *   Worker → Main:
 *     {type:"loaded", count:number, totalCount:number, chunk:number}
 *     {type:"filtered", indices:number[], total:number}
 *     {type:"stats", levels:object, sources:Array, timeRange:object, count:number}
 *     {type:"aggregated", buckets:Array}
 *     {type:"patterns", groups:Array}
 *     {type:"related", indices:number[]}
 *     {type:"fields", names:string[], topValues:object}
 */

"use strict";

// ─── Storage ───
const docs = [];         // Parsed log objects
const rawLines = [];     // Original JSON strings (for export)

// Inverted index: field → value → Set<lineIndex>
const index = new Map();

// ─── Index helpers ───
function addToIndex(field, value, lineIdx) {
  let fieldMap = index.get(field);
  if (!fieldMap) { fieldMap = new Map(); index.set(field, fieldMap); }
  let idxSet = fieldMap.get(value);
  if (!idxSet) { idxSet = []; fieldMap.set(value, idxSet); }
  idxSet.push(lineIdx);
}

// ─── Load & parse ───
function handleLoad(lines, chunk) {
  const startIdx = docs.length;
  for (let i = 0; i < lines.length; i++) {
    const raw = lines[i];
    try {
      const doc = JSON.parse(raw);
      const idx = startIdx + i;
      docs.push(doc);
      rawLines.push(raw);

      // Index all fields
      for (const [k, v] of Object.entries(doc)) {
        if (k === "ts" || k === "exc") continue; // Skip high-cardinality/large fields
        const sv = String(v);
        addToIndex(k, sv, idx);
      }
    } catch (e) {
      // Skip malformed lines
      docs.push(null);
      rawLines.push(raw);
    }
  }
  postMessage({ type: "loaded", count: lines.length, totalCount: docs.length, chunk });
}

// ─── Query parser ───
// Supports: level:error, src:orchestrator, field:>number, "bare words", AND, OR, NOT, parens
function parseQuery(query) {
  query = query.trim();
  if (!query) return null;

  const tokens = tokenize(query);
  return parseExpr(tokens, 0).node;
}

function tokenize(q) {
  const tokens = [];
  let i = 0;
  while (i < q.length) {
    if (q[i] === " " || q[i] === "\t") { i++; continue; }
    if (q[i] === "(") { tokens.push({ type: "LPAREN" }); i++; continue; }
    if (q[i] === ")") { tokens.push({ type: "RPAREN" }); i++; continue; }

    // Read a word or quoted string
    let word = "";
    if (q[i] === '"') {
      i++;
      while (i < q.length && q[i] !== '"') { word += q[i]; i++; }
      i++; // skip closing quote
    } else {
      while (i < q.length && q[i] !== " " && q[i] !== "\t" && q[i] !== "(" && q[i] !== ")") {
        word += q[i]; i++;
      }
    }

    if (word === "AND") tokens.push({ type: "AND" });
    else if (word === "OR") tokens.push({ type: "OR" });
    else if (word === "NOT") tokens.push({ type: "NOT" });
    else if (word.includes(":")) {
      const colon = word.indexOf(":");
      const field = word.slice(0, colon);
      let value = word.slice(colon + 1);
      // Strip surrounding quotes from value
      if (value.startsWith('"') && value.endsWith('"')) value = value.slice(1, -1);
      let op = "eq";
      if (value.startsWith(">")) { op = "gt"; value = value.slice(1); }
      else if (value.startsWith("<")) { op = "lt"; value = value.slice(1); }
      tokens.push({ type: "FIELD", field, value, op });
    } else {
      tokens.push({ type: "TEXT", value: word });
    }
  }
  return tokens;
}

function parseExpr(tokens, pos) {
  let { node, pos: p } = parseTerm(tokens, pos);
  while (p < tokens.length) {
    if (tokens[p].type === "OR") {
      p++;
      const right = parseTerm(tokens, p);
      node = { type: "OR", left: node, right: right.node };
      p = right.pos;
    } else if (tokens[p].type === "RPAREN") {
      break;
    } else {
      // Implicit AND
      if (tokens[p].type === "AND") p++;
      const right = parseTerm(tokens, p);
      node = { type: "AND", left: node, right: right.node };
      p = right.pos;
    }
  }
  return { node, pos: p };
}

function parseTerm(tokens, pos) {
  if (pos >= tokens.length) return { node: { type: "TRUE" }, pos };
  const t = tokens[pos];
  if (t.type === "NOT") {
    const inner = parseTerm(tokens, pos + 1);
    return { node: { type: "NOT", child: inner.node }, pos: inner.pos };
  }
  if (t.type === "LPAREN") {
    const inner = parseExpr(tokens, pos + 1);
    // Skip RPAREN
    const p = inner.pos < tokens.length && tokens[inner.pos].type === "RPAREN" ? inner.pos + 1 : inner.pos;
    return { node: inner.node, pos: p };
  }
  if (t.type === "FIELD") {
    return { node: { type: "FIELD", field: t.field, value: t.value, op: t.op }, pos: pos + 1 };
  }
  if (t.type === "TEXT") {
    return { node: { type: "TEXT", value: t.value }, pos: pos + 1 };
  }
  return { node: { type: "TRUE" }, pos: pos + 1 };
}

function matchDoc(node, doc, idx) {
  if (!node) return true;
  switch (node.type) {
    case "TRUE": return true;
    case "AND": return matchDoc(node.left, doc, idx) && matchDoc(node.right, doc, idx);
    case "OR": return matchDoc(node.left, doc, idx) || matchDoc(node.right, doc, idx);
    case "NOT": return !matchDoc(node.child, doc, idx);
    case "FIELD": {
      const docVal = doc[node.field];
      if (docVal === undefined) return false;
      const sv = String(docVal);
      if (node.op === "gt") return parseFloat(docVal) > parseFloat(node.value);
      if (node.op === "lt") return parseFloat(docVal) < parseFloat(node.value);
      // Case-insensitive match
      return sv.toLowerCase().includes(node.value.toLowerCase());
    }
    case "TEXT": {
      // Match against msg field (and all string fields as fallback)
      const msg = doc.msg || "";
      if (msg.toLowerCase().includes(node.value.toLowerCase())) return true;
      // Try regex
      try {
        const re = new RegExp(node.value, "i");
        if (re.test(msg)) return true;
      } catch (e) { /* not a valid regex */ }
      return false;
    }
  }
  return true;
}

// ─── Filter ───
function handleFilter(query) {
  const ast = parseQuery(query);
  const indices = [];
  for (let i = 0; i < docs.length; i++) {
    if (!docs[i]) continue;
    if (matchDoc(ast, docs[i], i)) indices.push(i);
  }
  postMessage({ type: "filtered", indices, total: docs.length });
}

// ─── Stats ───
function handleStats() {
  const levels = {};
  const sources = {};
  let firstTs = null, lastTs = null;
  for (let i = 0; i < docs.length; i++) {
    const d = docs[i];
    if (!d) continue;
    levels[d.level] = (levels[d.level] || 0) + 1;
    if (d.src) sources[d.src] = (sources[d.src] || 0) + 1;
    if (d.ts) {
      if (!firstTs) firstTs = d.ts;
      lastTs = d.ts;
    }
  }
  const topSources = Object.entries(sources).sort((a, b) => b[1] - a[1]).slice(0, 10);
  postMessage({ type: "stats", levels, sources: topSources, timeRange: { first: firstTs, last: lastTs }, count: docs.length });
}

// ─── Aggregate ───
function handleAggregate(field, bucket) {
  const buckets = new Map();
  for (let i = 0; i < docs.length; i++) {
    const d = docs[i];
    if (!d || !d.ts) continue;
    let key;
    if (bucket === "minute") {
      key = d.ts.slice(0, 16); // "2026-04-07T12:34"
    } else if (bucket === "hour") {
      key = d.ts.slice(0, 13); // "2026-04-07T12"
    } else {
      key = d.ts.slice(0, 10); // "2026-04-07"
    }
    let b = buckets.get(key);
    if (!b) { b = { key, total: 0, errors: 0 }; buckets.set(key, b); }
    b.total++;
    if (d.level === "ERROR" || d.level === "CRITICAL") b.errors++;
    // Track field value if specified
    if (field && field !== "level" && d[field] !== undefined) {
      if (!b.byField) b.byField = {};
      const fv = String(d[field]);
      b.byField[fv] = (b.byField[fv] || 0) + 1;
    }
  }
  postMessage({ type: "aggregated", buckets: Array.from(buckets.values()) });
}

// ─── Patterns ───
function handlePatterns() {
  // Group by message template (replace numbers/hex/uuids with placeholders)
  const groups = new Map();
  for (let i = 0; i < docs.length; i++) {
    const d = docs[i];
    if (!d || !d.msg) continue;
    // Normalize: replace numbers, UUIDs, hex strings with _
    const template = d.msg
      .replace(/[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}/gi, "_UUID_")
      .replace(/0x[0-9a-f]+/gi, "_HEX_")
      .replace(/\b\d+(\.\d+)?\b/g, "_N_");
    let g = groups.get(template);
    if (!g) { g = { template, count: 0, sample: d.msg, indices: [], levels: {} }; groups.set(template, g); }
    g.count++;
    g.indices.push(i);
    g.levels[d.level] = (g.levels[d.level] || 0) + 1;
  }
  // Sort by count descending, return top 100
  const sorted = Array.from(groups.values()).sort((a, b) => b.count - a.count).slice(0, 100);
  // Don't send full indices arrays for patterns (too large), just first 5
  const result = sorted.map(g => ({
    template: g.template,
    count: g.count,
    sample: g.sample,
    sampleIndices: g.indices.slice(0, 5),
    levels: g.levels,
  }));
  postMessage({ type: "patterns", groups: result });
}

// ─── Related ───
function handleRelated(lineIndex, windowSec) {
  const d = docs[lineIndex];
  if (!d) { postMessage({ type: "related", indices: [] }); return; }

  const ts = d.ts ? new Date(d.ts).getTime() : 0;
  const windowMs = (windowSec || 30) * 1000;
  const matchFields = ["task", "mission", "agent"].filter(f => d[f] !== undefined);

  if (matchFields.length === 0) {
    postMessage({ type: "related", indices: [] });
    return;
  }

  const indices = [];
  for (let i = 0; i < docs.length; i++) {
    if (i === lineIndex || !docs[i]) continue;
    const other = docs[i];
    // Must share at least one context field
    const shares = matchFields.some(f => other[f] !== undefined && String(other[f]) === String(d[f]));
    if (!shares) continue;
    // Must be within time window
    if (ts && other.ts) {
      const otherTs = new Date(other.ts).getTime();
      if (Math.abs(otherTs - ts) > windowMs) continue;
    }
    indices.push(i);
  }
  postMessage({ type: "related", indices });
}

// ─── Fields ───
function handleFields() {
  const names = Array.from(index.keys()).sort();
  const topValues = {};
  for (const [field, valMap] of index) {
    const entries = Array.from(valMap.entries())
      .sort((a, b) => b[1].length - a[1].length)
      .slice(0, 20);
    topValues[field] = entries.map(([v, idxs]) => ({ value: v, count: idxs.length }));
  }
  postMessage({ type: "fields", names, topValues });
}

// ─── Message dispatcher ───
onmessage = function(e) {
  const msg = e.data;
  switch (msg.type) {
    case "load":    handleLoad(msg.lines, msg.chunk); break;
    case "clear":   docs.length = 0; rawLines.length = 0; index.clear(); break;
    case "filter":  handleFilter(msg.query); break;
    case "stats":   handleStats(); break;
    case "aggregate": handleAggregate(msg.field, msg.bucket); break;
    case "patterns": handlePatterns(); break;
    case "related": handleRelated(msg.lineIndex, msg.windowSec); break;
    case "fields":  handleFields(); break;
  }
};
```

```javascript
// ─── Virtual Scroll ───
const ROW_HEIGHT = 28; // px, must match CSS .log-line height
const OVERSCAN = 20;   // extra rows above/below viewport

let filteredIndices = null; // null = show all, array = show only these
let selectedRow = -1;       // For keyboard nav

function getVisibleCount() {
  return Math.ceil($area.clientHeight / ROW_HEIGHT) + OVERSCAN * 2;
}

function getTotalRows() {
  return filteredIndices ? filteredIndices.length : allDocs.length;
}

// Pool of reusable row elements
const rowPool = [];
function ensurePool(n) {
  while (rowPool.length < n) {
    const row = document.createElement("div");
    row.className = "log-line";
    row.style.height = ROW_HEIGHT + "px";
    row.style.position = "absolute";
    row.style.left = "0";
    row.style.right = "0";
    rowPool.push(row);
    $area.appendChild(row);
  }
}

function renderVirtual() {
  const total = getTotalRows();
  $spacer.style.height = (total * ROW_HEIGHT) + "px";

  const scrollTop = $area.scrollTop;
  const startRow = Math.max(0, Math.floor(scrollTop / ROW_HEIGHT) - OVERSCAN);
  const endRow = Math.min(total, startRow + getVisibleCount());

  ensurePool(endRow - startRow);

  // Hide excess pool rows
  for (let i = endRow - startRow; i < rowPool.length; i++) {
    rowPool[i].style.display = "none";
  }

  for (let vi = startRow; vi < endRow; vi++) {
    const poolIdx = vi - startRow;
    const row = rowPool[poolIdx];
    const docIdx = filteredIndices ? filteredIndices[vi] : vi;
    const doc = allDocs[docIdx];

    row.style.display = "flex";
    row.style.top = (vi * ROW_HEIGHT) + "px";

    if (!doc) { row.innerHTML = ""; continue; }
    populateRow(row, doc, docIdx, vi === selectedRow);
  }
}
```

The `populateRow(row, doc, docIdx, selected)` function fills a row element with timestamp, level, source, message, and pills — same logic as the current `render()` but operating on a single recycled DOM node.

Full implementation is too large to inline here. The key architectural pieces:
- `$area` gets `position: relative; overflow-y: auto`
- `$spacer` is a child div that sets scroll height
- Row pool elements are `position: absolute` with `top` set per render
- `$area.onscroll = renderVirtual` drives the render loop
- Worker sends `filteredIndices` which the virtual scroller uses directly

```javascript
  if (!buckets || buckets.length === 0) return;

  const maxTotal = Math.max(...buckets.map(b => b.total));
  if (maxTotal === 0) return;

  const barW = Math.max(1, w / buckets.length);

  for (let i = 0; i < buckets.length; i++) {
    const b = buckets[i];
    const x = i * barW;
    const barH = (b.total / maxTotal) * (h - 4);

    // Base bar — accent color with opacity
    ctx.fillStyle = getComputedStyle(document.documentElement).getPropertyValue("--accent").trim() + "40";
    ctx.fillRect(x, h - barH, barW - 0.5, barH);

    // Error overlay — red
    if (b.errors > 0) {
      const errH = (b.errors / maxTotal) * (h - 4);
      ctx.fillStyle = getComputedStyle(document.documentElement).getPropertyValue("--error").trim();
      ctx.fillRect(x, h - errH, barW - 0.5, errH);
    }
  }
}

function renderMinimap() {
  const canvas = $minimap;
  const ctx = canvas.getContext("2d");
  canvas.width = 12;
  canvas.height = canvas.clientHeight;
  const h = canvas.height;
  ctx.clearRect(0, 0, 12, h);

  const total = getTotalRows();
  if (total === 0) return;

  const errorColor = getComputedStyle(document.documentElement).getPropertyValue("--error").trim();
  const warnColor = getComputedStyle(document.documentElement).getPropertyValue("--warn").trim();

  for (let i = 0; i < total; i++) {
    const docIdx = filteredIndices ? filteredIndices[i] : i;
    const doc = allDocs[docIdx];
    if (!doc) continue;
    const y = (i / total) * h;
    if (doc.level === "ERROR" || doc.level === "CRITICAL") {
      ctx.fillStyle = errorColor;
      ctx.fillRect(0, y, 12, Math.max(2, h / total));
    } else if (doc.level === "WARNING") {
      ctx.fillStyle = warnColor;
      ctx.fillRect(0, y, 6, Math.max(1, h / total));
    }
  }
}
```

```javascript
function syntaxHighlightJSON(obj, indent) {
  indent = indent || 0;
  const pad = "  ".repeat(indent);
  let html = "";
  if (typeof obj === "object" && obj !== null && !Array.isArray(obj)) {
    html += "{\n";
    const entries = Object.entries(obj);
    entries.forEach(([k, v], i) => {
      html += `${pad}  <span class="json-key">"${escHtml(k)}"</span>: `;
      html += syntaxHighlightJSON(v, indent + 1);
      if (i < entries.length - 1) html += ",";
      html += "\n";
    });
    html += pad + "}";
  } else if (Array.isArray(obj)) {
    html += "[" + obj.map(v => syntaxHighlightJSON(v, indent + 1)).join(", ") + "]";

  } else if (typeof obj === "boolean") {
    html += `<span class="json-bool">${obj}</span>`;
  } else {

  }
  return html;
}
```

```javascript
// ─── Export ───
function exportLogs(format) {
  const indices = filteredIndices || allDocs.map((_, i) => i);
  const docs = indices.map(i => allDocs[i]).filter(Boolean);

  let content, filename, mime;
  switch (format) {
    case "jsonl":
      content = docs.map(d => JSON.stringify(d)).join("\n");
      filename = "yazbunu-export.jsonl";
      mime = "application/jsonl";
      break;
    case "csv": {
      // Collect all unique keys
      const keys = new Set();
      docs.forEach(d => Object.keys(d).forEach(k => keys.add(k)));
      const cols = Array.from(keys);
      const header = cols.map(c => `"${c}"`).join(",");
      const rows = docs.map(d => cols.map(c => {
        const v = d[c];
        if (v === undefined) return "";
        return `"${String(v).replace(/"/g, '""')}"`;
      }).join(","));
      content = header + "\n" + rows.join("\n");
      filename = "yazbunu-export.csv";
      mime = "text/csv";
      break;
    }
    case "markdown": {
      const cols = ["ts", "level", "src", "msg"];
      content = "| " + cols.join(" | ") + " |\n";
      content += "| " + cols.map(() => "---").join(" | ") + " |\n";
      docs.forEach(d => {
        content += "| " + cols.map(c => String(d[c] || "").replace(/\|/g, "\\|")).join(" | ") + " |\n";
      });
      filename = "yazbunu-export.md";
      mime = "text/markdown";
      break;
    }
    default: // text
      content = docs.map(d => `${d.ts} ${d.level} ${d.src}: ${d.msg}`).join("\n");
      filename = "yazbunu-export.txt";
      mime = "text/plain";
  }

  downloadText(filename, content, mime);
}

function downloadText(filename, content, mime) {
  mime = mime || "text/plain";
  const blob = new Blob([content], { type: mime });
  const url = URL.createObjectURL(blob);
  const a = document.createElement("a");
  a.href = url;
  a.download = filename;
  a.click();
  URL.revokeObjectURL(url);
}

function copyToClipboard(format) {
  const indices = filteredIndices || allDocs.map((_, i) => i);
  const docs = indices.map(i => allDocs[i]).filter(Boolean);

  } else {
    text = docs.map(d => `${d.ts} ${d.level} ${d.src}: ${d.msg}`).join("\n");
  }
  navigator.clipboard.writeText(text);
  $status.textContent = `Copied ${docs.length} lines`;
  setTimeout(() => { $status.textContent = ""; }, 2000);
}
```

## Task 17: Browser — PWA Service Worker

Offline app shell caching for instant launch.

**Files:**
- Create: `packages/yazbunu/src/yazbunu/static/sw.js`
- Modify: `packages/yazbunu/src/yazbunu/static/manifest.json`

- [ ] **Step 1: Create service worker**

Create `packages/yazbunu/src/yazbunu/static/sw.js`:

```javascript
const CACHE_NAME = "yazbunu-v2";
const SHELL_FILES = [
  "/",
  "/static/viewer.html",
  "/static/worker.js",
  "/static/themes.js",
  "/static/manifest.json",
];

self.addEventListener("install", (e) => {
  e.waitUntil(
    caches.open(CACHE_NAME).then(cache => cache.addAll(SHELL_FILES))
  );
  self.skipWaiting();
});

self.addEventListener("activate", (e) => {
  e.waitUntil(
    caches.keys().then(names =>
      Promise.all(names.filter(n => n !== CACHE_NAME).map(n => caches.delete(n)))
    )
  );
  self.clients.claim();
});

self.addEventListener("fetch", (e) => {
  const url = new URL(e.request.url);

  // API calls and WebSocket: always network
  if (url.pathname.startsWith("/api/") || url.pathname.startsWith("/ws/") || url.pathname === "/health") {
    return;
  }

  // App shell: cache-first, then network
  e.respondWith(
    caches.match(e.request).then(cached => cached || fetch(e.request))
  );
});
```
